/* EMERGENCY RESET SERVICE WORKER (standby — not active).
   If a deploy ever leaves phones stuck on a broken cached version:
   replace the live service worker file (sw.js / audit_sw.js) with the
   contents of this file and push. It deletes all caches, unregisters
   itself, and reloads every open page, giving clients a clean slate.
   After clients recover, ship a normal service worker again. */
self.addEventListener('install', () => { self.skipWaiting(); });
self.addEventListener('activate', e => {
  e.waitUntil((async () => {
    const keys = await caches.keys();
    await Promise.all(keys.map(k => caches.delete(k)));
    await self.registration.unregister();
    const clients = await self.clients.matchAll({ type: 'window' });
    clients.forEach(c => c.navigate(c.url));
  })());
});
